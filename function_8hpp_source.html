48template<typename ... Args>
49struct context
50{
51 template<std::size_t N>
52 using element = mp11::mp_take_c<mp11::mp_list<Args...>, N>;
53
55 template<std::size_t Idx>
56 void set(element<Idx> value)
57 {
58 sqlite3_set_auxdata(ctx_, Idx, *static_cast<void**>(&value),
59 new element<Idx>(std::move(value)),
60 +[](void * ptr)
61 {
62 delete static_cast<element<Idx> *>(ptr);
63 });
64 }
65
67 template<std::size_t Idx>
68 auto get() -> element<Idx> &
69 {
70 using type = element<Idx> ;
71 auto p = static_cast<type*>(sqlite3_get_auxdata(ctx_, Idx));
72 if (p == nullptr)
73 throw_exception(std::invalid_argument("argument not set"));
74 return p;
75 }
76
78 template<std::size_t Idx>
79 auto get_if() -> element<Idx> *
80 {
81 using type = element<Idx> ;
82 return static_cast<type*>(sqlite3_get_auxdata(ctx_, Idx));
83 }
84
85
86 explicit context(sqlite3_context * ctx) noexcept : ctx_(ctx) {}
87
89 template<typename T>
90 auto set_result(T && val)
91#if !defined(BOOST_SQLITE_GENERATING_DOCS)
92 -> decltype(detail::set_result(static_cast<sqlite3_context*>(nullptr), std::forward<T>(val)))
93#endif
94 {
95 detail::set_result(ctx_, std::forward<T>(val));
96 }
98 void set_error(const char * message, int code = SQLITE_ERROR)
99 {
100 sqlite3_result_error(ctx_, message, code);
101 }
103 connection get_connection() const
104 {
105 return connection{sqlite3_context_db_handle(ctx_), false};
106 }
107
108 private:
109 sqlite3_context * ctx_;
110};

116template<typename Func, typename ... Args, std::size_t Extent>
117auto create_scalar_function_impl(sqlite3 * db,
118 const std::string & name,
119 Func && func,
120 std::tuple<context<Args...>, boost::span<value, Extent>> * ,
121 std::false_type /* void return */,
122 std::false_type /* is pointer */) -> int
123{
124 using func_type = typename std::decay<Func>::type;
125 return sqlite3_create_function_v2(
126 db, name.c_str(),
127 Extent == boost::dynamic_extent ? -1 : static_cast<int>(Extent),
128 SQLITE_UTF8,
129 new func_type(std::forward<Func>(func)),
130 +[](sqlite3_context* ctx, int len, sqlite3_value** args)
131 {
132 auto cc = context<Args...>(ctx);
133 auto aa = reinterpret_cast<value*>(args);
134 auto &f = *reinterpret_cast<func_type*>(sqlite3_user_data(ctx));
135
136 try
137 {
138 set_result(ctx, f(cc, boost::span<value, Extent>{aa, static_cast<std::size_t>(len)}));
139 }
140 BOOST_SQLITE_CATCH_RESULT(ctx)
141
142 }, nullptr, nullptr,
143 +[](void * ptr) noexcept {delete static_cast<func_type*>(ptr);}
144 );
145}

259template<typename Func>
260int create_aggregate_function(sqlite3 * db, const std::string & name, Func && func)
261{
262 using args_type = callable_traits::args_t<decltype(&Func::step)>;
263 using result_type = callable_traits::return_type<decltype(&Func::final)>;
264 using context_type = typename std::remove_reference<typename std::tuple_element<1u, args_type>::type>::type;
265 using span_type = typename std::tuple_element<2U, args_type>::type;
266 using func_type = typename std::decay<Func>::type;
267
268 return sqlite3_create_function_v2(
269 db, name.c_str(),
270 span_type::extent == boost::dynamic_extent ? -1 : static_cast<int>(span_type::extent),
271 SQLITE_UTF8,
272 new func_type(std::forward<Func>(func)),
273 nullptr,
274 +[](sqlite3_context* ctx, int len, sqlite3_value** args)
275 {
276 auto aa = reinterpret_cast<value*>(args);
277 auto f = reinterpret_cast<Func*>(sqlite3_user_data(ctx));
278 auto c = static_cast<context_type*>(sqlite3_aggregate_context(ctx, 0));
279
280 try
281 {
282 if (c == nullptr)
283 {
284 auto p = sqlite3_aggregate_context(ctx, sizeof(context_type));
285 c = new (p) context_type();
286 }
287 f->step(*c, span_type{aa, static_cast<std::size_t>(len)});
288 }
289 BOOST_SQLITE_CATCH_RESULT(ctx)
290 },
291 [](sqlite3_context* ctx)
292 {
293 auto f = reinterpret_cast<Func*>(sqlite3_user_data(ctx));
294 auto c = static_cast<context_type*>(sqlite3_aggregate_context(ctx, 0));
295
296 try
297 {
298 if (c == nullptr)
299 {
300 auto p = sqlite3_aggregate_context(ctx, sizeof(context_type));
301 c = new (p) context_type();
302 }
303 set_result(ctx, f->final(*c));
304 c->~context_type();
305 }
306 BOOST_SQLITE_CATCH_RESULT(ctx)
307 },
308 [](void * ptr) noexcept { delete static_cast<func_type*>(ptr);}
309 );
310}
