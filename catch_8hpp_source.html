13#define BOOST_SQLITE_CATCH_RESULT(ctx) \
14catch (boost::system::system_error & se) \
15{ \
16 if (se.code().category() == boost::sqlite::sqlite_category()) \
17 sqlite3_result_error_code(ctx, se.code().value()); \
18 sqlite3_result_error(ctx, se.what(), std::strlen(se.what())); \
19} \
20catch(std::bad_alloc &) { sqlite3_result_error_nomem(ctx); } \
21catch(std::length_error &) { sqlite3_result_error_toobig(ctx); } \
22catch(std::out_of_range &) { sqlite3_result_error_code(ctx, SQLITE_RANGE);} \
23catch(std::logic_error &le) \
24{ \
25 sqlite3_result_error_code(ctx, SQLITE_MISUSE); \
26 sqlite3_result_error(ctx, le.what(), std::strlen(le.what())); \
27} \
28catch(std::exception & ex) \
29{ \
30 sqlite3_result_error(ctx, ex.what(), std::strlen(ex.what())); \
31} \
32catch(...) {sqlite3_result_error_code(ctx, SQLITE_ERROR); }

34#define BOOST_SQLITE_CATCH_ASSIGN_STR_AND_RETURN(msg) \
35catch (boost::system::system_error & se) \
36{ \
37 auto code = SQLITE_ERROR; \
38 if (se.code().category() == boost::sqlite::sqlite_category()) \
39 code = se.code().value(); \
40 msg = sqlite3_mprintf("%s", se.what()); \
41 return code; \
42} \
43catch(std::bad_alloc &) { return SQLITE_NOMEM; } \
44catch(std::length_error &) { return SQLITE_TOOBIG; } \
45catch(std::out_of_range &) { return SQLITE_RANGE;} \
46catch(std::logic_error &le) \
47{ \
48 msg = sqlite3_mprintf("%s", le.what()); \
49 return SQLITE_MISUSE; \
50} \
51catch(std::exception & ex) \
52{ \
53 msg = sqlite3_mprintf("%s", ex.what()); \
54 return SQLITE_ERROR; \
55} \
56catch(...) {return SQLITE_ERROR; } \
