100 struct visitor
101 {
102 sqlite3_stmt * stmt;
103 int col;
104
105 int operator()(variant2::monostate )
106 {
107 return sqlite3_bind_null(stmt, col);
108 }
109 int operator()(int i )
110 {
111 return sqlite3_bind_int(stmt, col, i);
112 }
113 int operator()(sqlite3_int64 i64 )
114 {
115 return sqlite3_bind_int64(stmt, col, i64);
116 }
117
118 int operator()(blob_view blob)
119 {
120 if (blob.size() > static_cast<std::size_t>(std::numeric_limits<int>::max()))
121 return sqlite3_bind_blob64(stmt, col, blob.data(), blob.size(), SQLITE_STATIC);
122 else
123 return sqlite3_bind_blob(stmt, col, blob.data(), blob.size(), SQLITE_STATIC);
124 }
125
126 int operator()(string_view text)
127 {
128 if (text.size() > std::numeric_limits<int>::max())
129 return sqlite3_bind_text64(stmt, col, text.data(), text.size(), SQLITE_STATIC, SQLITE_UTF8);
130 else
131 return sqlite3_bind_text(stmt, col, text.data(), text.size(), SQLITE_STATIC);
132 }
133 int operator()(double value)
134 {
135 return sqlite3_bind_double(stmt, col, value);
136 }
137 int operator()(zero_blob zb)
138 {
139 if (static_cast<std::size_t>(zb) > static_cast<std::size_t>(std::numeric_limits<int>::max()))
140 return sqlite3_bind_zeroblob64(stmt, col, static_cast<sqlite3_uint64>(zb));
141 else
142 return sqlite3_bind_zeroblob(stmt, col, static_cast<int>(zb));
143 }
144 int operator()(std::pair<std::unique_ptr<void, void(*)(void*)>, const char*> & p)
145 {
146 auto d =p.first.get_deleter();
147 return sqlite3_bind_pointer(stmt, col, p.first.release(), p.second, d);
148 }
149 };
150
151 mutable // so we can use it with
152 variant2::variant<variant2::monostate, int, sqlite3_int64,
153 blob_view, string_view, double, zero_blob,
154 std::pair<std::unique_ptr<void, void(*)(void*)>, const char*>> impl_;
155};

320 template<typename ... Args>
321 void bind_impl(std::tuple<Args...> && vec,
322 error_code & ec,
323 error_info & ei)
324 {
325 const auto sz = sqlite3_bind_parameter_count(impl_.get());
326 if (sizeof...(Args) < sz)
327 {
328 BOOST_SQLITE_ASSIGN_EC(ec, SQLITE_ERROR);
329 ei.set_message("To few parameters provided. Needs " + std::to_string(sz)
330 + " got " + std::to_string(sizeof...(Args)));
331 return;
332 }
333
334 int i = 1, ar = SQLITE_OK;
335 mp11::tuple_for_each(std::move(vec),
336 [&](param_ref pr)
337 {
338 if (ar == SQLITE_OK)
339 ar = pr.apply(impl_.get(), i++);
340 });
341 if (ar != SQLITE_OK)
342 {
343 BOOST_SQLITE_ASSIGN_EC(ec, ar);
344 ei.set_message(sqlite3_errmsg(sqlite3_db_handle(impl_.get())));
345 return;
346 }
347 }

405 template<typename ParamMap>
406 void bind_impl(ParamMap && vec, error_code & ec, error_info & ei,
407 typename std::enable_if<
408 std::is_convertible<typename std::decay<ParamMap>::type::key_type, string_view>::value &&
409 std::is_convertible<typename std::decay<ParamMap>::type::mapped_type, param_ref>::value
410 >::type * = nullptr)
411 {
412 for (auto i = 1; i <= sqlite3_bind_parameter_count(impl_.get()); i ++)
413 {
414 auto c = sqlite3_bind_parameter_name(impl_.get(), i);
415 if (c == nullptr)
416 {
417 BOOST_SQLITE_ASSIGN_EC(ec, SQLITE_MISUSE);
418 ei.set_message("Parameter maps require all parameters to be named.");
419 return ;
420 }
421 auto itr = vec.find(c+1);
422 if (itr == vec.end())
423 {
424 BOOST_SQLITE_ASSIGN_EC(ec, SQLITE_MISUSE);
425 ei.set_message("Can't find value for key '" + std::string(c+1) + "'");
426 return ;
427 }
428 int ar = SQLITE_OK;
429 if (std::is_rvalue_reference<ParamMap&&>::value)
430 ar = param_ref(std::move(itr->second)).apply(impl_.get(), i);
431 else
432 ar = param_ref(itr->second).apply(impl_.get(), i);
433
434 if (ar != SQLITE_OK)
435 {
436
437 BOOST_SQLITE_ASSIGN_EC(ec, ar);
438 ei.set_message(sqlite3_errmsg(sqlite3_db_handle(impl_.get())));
439 return;
440 }
441 }

444 void bind_impl(std::initializer_list<std::pair<string_view, param_ref>> params,
445 error_code & ec, error_info & ei)
446 {
447 for (auto i = 1; i <= sqlite3_bind_parameter_count(impl_.get()); i ++)
448 {
449 auto c = sqlite3_bind_parameter_name(impl_.get(), i);
450 if (c == nullptr)
451 {
452 BOOST_SQLITE_ASSIGN_EC(ec, SQLITE_MISUSE);
453 ei.set_message("Parameter maps require all parameters to be named.");
454 return ;
455 }
456
457 auto itr = std::find_if(params.begin(), params.end(),
458 [&](const std::pair<string_view, param_ref> & p)
459 {
460 return p.first == (c+1);
461 });
462 if (itr == params.end())
463 {
464 BOOST_SQLITE_ASSIGN_EC(ec, SQLITE_MISUSE);
465 ei.set_message("Can't find value for key '" + std::string(c+1) + "'");
466 return ;
467 }
468 auto ar = itr->second.apply(impl_.get(), i);
469 if (ar != SQLITE_OK)
470 {
471
472 BOOST_SQLITE_ASSIGN_EC(ec, ar);
473 ei.set_message(sqlite3_errmsg(sqlite3_db_handle(impl_.get())));
474 return;
475 }
