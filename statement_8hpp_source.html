18struct binder
19{
20 sqlite3_stmt * stm_;
21 int col;
22
23 int operator()(variant2::monostate)
24 {
25 return sqlite3_bind_null(stm_, col);
26 }
27 int operator()(std::nullptr_t)
28 {
29 return sqlite3_bind_null(stm_, col);
30 }
31 template<typename I,
32 typename = std::enable_if_t<std::is_integral<I>::value>>
33 int operator()(I value)
34 {
35 BOOST_IF_CONSTEXPR ((sizeof(I) == sizeof(int) && std::is_unsigned<I>::value)
36 || (sizeof(I) > sizeof(int)))
37 return sqlite3_bind_int64(stm_, col, static_cast<sqlite3_int64>(value));
38 else
39 return sqlite3_bind_int(stm_, col, static_cast<int>(value));
40 }
41 int operator()(blob_view blob)
42 {
43 if (blob.size() > std::numeric_limits<int>::max())
44 return sqlite3_bind_blob(stm_, col, blob.data(), blob.size(), nullptr);
45 else
46 return sqlite3_bind_blob64(stm_, col, blob.data(), blob.size(), nullptr);
47 }
48
49 int operator()(string_view text)
50 {
51 if (text.size() > std::numeric_limits<int>::max())
52 return sqlite3_bind_text(stm_, col, text.data(), text.size(), nullptr);
53 else
54 return sqlite3_bind_text64(stm_, col, text.data(), text.size(), nullptr, SQLITE_UTF8);
55 }
56
57 int operator()(double value)
58 {
59 return sqlite3_bind_double(stm_, col, value);
60 }
61
62 template<typename T>
63 int operator()(std::unique_ptr<T> ptr)
64 {
65 return sqlite3_bind_pointer(stm_, col, ptr.release(), typeid(T).name(),
66 +[](void * ptr){delete static_cast<T*>(ptr);});
67 }
68
69 template<typename T>
70 int operator()(std::unique_ptr<T, void(*)(T*)> ptr)
71 {
72 return sqlite3_bind_pointer(stm_, col, ptr.release(), typeid(T).name(),
73 static_cast<void(*)(void*)>(ptr.get_deleter()));
74 }
75
76 template<typename T, typename Deleter>
77 auto operator()(std::unique_ptr<T, Deleter> ptr)
78 -> typename std::enable_if<std::is_empty<Deleter>::value &&
79 std::is_default_constructible<Deleter>::value, int>::type
80 {
81 return sqlite3_bind_pointer(stm_, col, ptr.release(), typeid(T).name(),
82 +[](void * ptr){Deleter()(static_cast<T*>(ptr));});
83 }

235 template<typename T>
236 bool bind_step(
237 T && value,
238 std::size_t idx,
239 error_code & ec,
240 error_info & ei)
241 {
242 if (!ec)
243 {
244 auto cc = detail::binder{impl_.get(), static_cast<int>(idx)}(std::forward<T>(value));
245 if (cc != SQLITE_OK)
246 {
247 BOOST_SQLITE_ASSIGN_EC(ec, cc);
248 ei.set_message(sqlite3_errmsg(sqlite3_db_handle(impl_.get())));
249 }
250 }
251 return !ec;
252 }

254 template<typename ... Args, std::size_t ... Idx>
255 void bind_impl(std::tuple<Args...> && tupl,
256 error_code & ec,
257 error_info & ei,
258 std::index_sequence<Idx...>)
259 {
260 const auto sz = sqlite3_bind_parameter_count(impl_.get());
261 if (sizeof...(Idx) < sz)
262 {
263 BOOST_SQLITE_ASSIGN_EC(ec, SQLITE_ERROR);
264 ei.set_message("To few parameters provided. Needs " + std::to_string(sz)
265 + " got " + std::to_string(sizeof...(Idx)));
266 }
267
268 boost::ignore_unused(bind_step(std::move(std::get<Idx>(tupl)), Idx + 1, ec, ei)...);
269 }
